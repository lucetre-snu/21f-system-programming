84 static int dirent_compare(const void *a, const void *b)
85 {
86  struct dirent *e1 = (struct dirent*)a;
87  struct dirent *e2 = (struct dirent*)b;
88 
89  // if one of the entries is a directory, it comes first
90  if (e1->d_type != e2->d_type) {
91  if (e1->d_type == DT_DIR) return -1;
92  if (e2->d_type == DT_DIR) return 1;
93  }
94 
95  // otherwise sorty by name
96  return strcmp(e1->d_name, e2->d_name);

106 void processDir(const char *dn, const char *pstr, struct summary *stats, unsigned int flags)
107 {
108  DIR *d = opendir(dn);
109  // erro handling
110  if (errno == EACCES) {
111  printf((flags & F_TREE) ? "%s`-%s\n" : "%s %s\n", pstr, "ERROR: Permission denied");
112  return;
113  }
114  else if (errno == ENOTDIR) {
115  printf(" ERROR: Not a directory\n");
116  return;
117  }
118  else if (d == NULL) {
119  printf(" ERROR: No such file or directory\n");
120  return;
121  }
122  struct dirent *entry;
123 
124  // determine number of entries before storing entries in dir for dynamic mem allocation
125  unsigned int nent = 0;
126  while((entry = getNext(d)) != NULL) {
127  nent++;
128  }
129  struct dirent *entries = malloc(sizeof(struct dirent)*nent);
130 
131  // store every entries in dirent array
132  int i = 0;
133  d = opendir(dn);
134  while((entry = getNext(d)) != NULL) {
135  entries[i++] = *entry;
136  }
137 
138  // sort in order of dir and filename using dirent comparator
139  qsort(entries, nent, sizeof(entries[0]), dirent_compare);
140 
141  for (unsigned int pos = 0; pos < nent; pos++) {
142  struct dirent *this = &entries[pos];
143 
144  char *fn, *out, *tmp;
145  // handling path string
146  if (asprintf(&fn, "%s/%s", dn, this->d_name) == -1) panic("OUT OF MEMORY!");
147 
148  // handling output prefix format string
149  if (flags & F_TREE) {
150  if (pos == nent - 1) {
151  if (asprintf(&out, "%s", pstr) == -1) panic("OUT OF MEMORY!");
152  if (asprintf(&tmp, "%s`-%s", out, this->d_name) == -1) panic("OUT OF MEMORY!");
153  if (asprintf(&out, "%s ", out) == -1) panic("OUT OF MEMORY!");
154  } else {
155  if (asprintf(&out, "%s|", pstr) == -1) panic("OUT OF MEMORY!");
156  if (asprintf(&tmp, "%s-%s", out, this->d_name) == -1) panic("OUT OF MEMORY!");
157  if (asprintf(&out, "%s ", out) == -1) panic("OUT OF MEMORY!");
158  }
159  } else {
160  if (asprintf(&out, "%s ", pstr) == -1) panic("OUT OF MEMORY!");
161  if (asprintf(&tmp, "%s%s", out, this->d_name) == -1) panic("OUT OF MEMORY!");
162  }
163 
164  if (flags & F_VERBOSE) {
165  struct stat st;
166  if (lstat(fn, &st) == -1) panic(" ERROR: No such file or directory");
167 
168  if (strlen(tmp) > 54) {
169  strncpy(tmp + 51, "...\0", 4);
170  }
171  printf("%-54s ", tmp);
172 
173  unsigned char type = (this->d_type == DT_REG) ? ' ' :
174  (this->d_type == DT_DIR) ? 'd' :
175  (this->d_type == DT_LNK) ? 'l' :
176  (this->d_type == DT_CHR) ? 'c' :
177  (this->d_type == DT_BLK) ? 'b' :
178  (this->d_type == DT_FIFO) ? 'f' :
179  (this->d_type == DT_SOCK) ? 's' : '?';
180 
181  // if directory type is unknown
182  if (type == '?') {
183  printf("File type could not be determined\n");
184  free(fn);
185  free(out);
186  free(tmp);
187  continue;
188  }
189 
190  // accumulate number of files/dirs/links/pipes/sockets/blocks and size
191  stats->dirs += (type == 'd');
192  stats->files += (type == ' ');
193  stats->links += (type == 'l');
194  stats->fifos += (type == 'f');
195  stats->socks += (type == 's');
196  stats->size += st.st_size;
197  stats->blocks += st.st_blocks;
198 
199  // extract user/group id from stat and find corresponding user/group name
200  printf("%8s:%-8s %10ld %8ld %c\n",
201  getpwuid(st.st_uid)->pw_name,
202  getgrgid(st.st_gid)->gr_name,
203  st.st_size,
204  st.st_blocks,
205  type);
206 
207  } else {
208  printf("%s\n", tmp);
209  }
210  if (this->d_type == DT_DIR) {
211  // recurse with relative path, output prefix format string, accumulated stats, flag mode
212  processDir(fn, out, stats, flags);
213  }
214  free(fn);
215  free(out);
216  free(tmp);
217  }
218 }

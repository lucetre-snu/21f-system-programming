11 // Dynamic memory manager
12 // ======================
13 // This module implements a custom dynamic memory manager.
14 //
15 // Heap organization:
16 // ------------------
17 // The data segment for the heap is provided by the dataseg module. A 'word' in the heap is
18 // eight bytes.
19 //
20 // Implicit free list:
21 // -------------------
22 // - minimal block size: 32 bytes (header +footer + 2 data words)
23 // - h,f: header/footer of free block
24 // - H,F: header/footer of allocated block
25 //
26 // - state after initialization
27 //
28 // initial sentinel half-block end sentinel half-block
29 // | |
30 // ds_heap_start | heap_start heap_end ds_heap_brk
31 // | | | | |
32 // v v v v v
33 // +---+---+-----------------------------------------+---+---+
34 // |???| F | h : : f | H |???|
35 // +---+---+-----------------------------------------+---+---+
36 // ^ ^
37 // | |
38 // 32-byte aligned 32-byte aligned
39 //
40 // - allocation policies: first, next, best fit
41 // - block splitting: always at 32-byte boundaries
42 // - immediate coalescing upon free
43 //

173 void mm_init(AllocationPolicy ap)
174 {
175  LOG(1, "mm_init()");
176 
177  //
178  // set allocation policy
179  //
180  char *apstr;
181  switch (ap) {
182  case ap_FirstFit: get_free_block = ff_get_free_block; apstr = "first fit"; break;
183  case ap_NextFit: get_free_block = nf_get_free_block; apstr = "next fit"; break;
184  case ap_BestFit: get_free_block = bf_get_free_block; apstr = "best fit"; break;
185  default: PANIC("Invalid allocation policy.");
186  }
187  LOG(2, " allocation policy %s\n", apstr);
188 
189  //
190  // retrieve heap status and perform a few initial sanity checks
191  //
192  ds_heap_stat(&ds_heap_start, &ds_heap_brk, NULL);
193  PAGESIZE = ds_getpagesize();
194 
195  LOG(2, " ds_heap_start: %p\n"
196  " ds_heap_brk: %p\n"
197  " PAGESIZE: %d\n",
198  ds_heap_start, ds_heap_brk, PAGESIZE);
199 
200  if (ds_heap_start == NULL) PANIC("Data segment not initialized.");
201  if (ds_heap_start != ds_heap_brk) PANIC("Heap not clean.");
202  if (PAGESIZE == 0) PANIC("Reported pagesize == 0.");
203 
204  //
205  // initialize heap
206  //
207  ds_sbrk(INITCHUNK);
208  ds_heap_brk = ds_sbrk(0);
209  LOG(2, " allocated memory, new ds_heap_brk is at %p", ds_heap_brk);
210  if (ds_heap_brk == (void*)-1) PANIC("Cannot extend heap.");
211 
212  heap_start = ds_heap_start;
213  LOG(2, " old heap_start is at %p", heap_start);
214  next_block = heap_start = (TYPE*)(((TYPE)heap_start + BS) / BS * BS);
215  LOG(2, " new heap_start is at %p", heap_start);
216  PUT(PREV_PTR(heap_start), PACK(0, ALLOC));
217 
218  heap_end = ds_heap_brk;
219  LOG(2, " old heap_end is at %p", heap_end);
220  heap_end = (TYPE*)(((TYPE)heap_end - 1) / BS * BS);
221  LOG(2, " new heap_start is at %p", heap_end);
222  PUT(heap_end, PACK(0, ALLOC));
223 
224  size_t size = heap_end - heap_start;
225  TYPE bdrtag = PACK(size, FREE);
226  PUT(heap_start, bdrtag);
227  PUT(PREV_PTR(heap_end), bdrtag);
228 
229  //
230  // heap is initialized
231  //
232  mm_initialized = 1;
233 }

236 void* mm_malloc(size_t size)
237 {
238  LOG(1, "mm_malloc(0x%lx (%ld))", size, size);
239 
240  assert(mm_initialized);
241  size_t req_size = ceil((double)(size + 2*TYPE_SIZE) / BS) * BS;
242 
243  void *start_alloc = get_free_block(req_size);
244  if (start_alloc == NULL) {
245  LOG(1, "expand_heap(0x%lx (%ld))", req_size, req_size);
246 
247  void *p = heap_end;
248  size_t size = (GET_STATUS(PREV_PTR(p)) == FREE) ? GET_SIZE(PREV_PTR(p)) : 0;
249  LOG(2, " last block");
250  LOG(2, " header %p", START(PREV_PTR(p)));
251  LOG(2, " footer %p", PREV_PTR(p));
252  LOG(2, " size %lx (%ld)", GET_SIZE(PREV_PTR(p)), GET_SIZE(PREV_PTR(p)));
253  LOG(2, " status %d\n", GET_STATUS(PREV_PTR(p)));
254 
255 
256  size_t chunk_size = ceil((double)(req_size - size + 1) / CHUNKSIZE) * CHUNKSIZE;
257  if (chunk_size < INITCHUNK)
258  chunk_size = INITCHUNK;
259 
260  LOG(2, " increment heap by 0x%lx (%ld) bytes", chunk_size, chunk_size);
261  LOG(1, "ds_sbrk(+0x%lx)", chunk_size);
262  ds_sbrk(chunk_size);
263  heap_end = ds_heap_brk = ds_sbrk(0);
264  heap_end = (TYPE*)(((TYPE)heap_end - 1) / BS * BS);
265 
266  LOG(1, "ds_sbrk(+0x%lx)", 0);
267  LOG(2, " new heap_end at %p", heap_end);
268 
269  LOG(1, "coalesce()");
270  LOG(2, " coalescing with preceeding block");
271 
272  size += (size_t)(heap_end - p);
273  start_alloc = heap_end - size;
274  LOG(2, " last block now at %p with size 0x%lx (%ld) bytes", start_alloc, size, size);
275 
276  TYPE bdrtag = PACK(size, FREE);
277  PUT(start_alloc, bdrtag);
278  PUT(PREV_PTR(heap_end), bdrtag);
279  PUT(heap_end, PACK(0, ALLOC));
280  }
281  size_t prev_size = GET_SIZE(start_alloc);
282  void *end_alloc = PREV_PTR(start_alloc + req_size);
283 
284  TYPE bdrtag_alloc = PACK(req_size, ALLOC);
285  PUT(start_alloc, bdrtag_alloc);
286  PUT(end_alloc, bdrtag_alloc);
287 
288  size_t cur_size = prev_size - req_size;
289  if (cur_size) {
290  void *start_free = start_alloc + req_size;
291  void *end_free = PREV_PTR(start_free + cur_size);
292 
293  TYPE bdrtag_free = PACK(cur_size, FREE);
294  PUT(start_free, bdrtag_free);
295  PUT(end_free, bdrtag_free);
296  }
297 
298  return start_alloc + TYPE_SIZE;
299 }

342 void mm_free(void *ptr)
343 {
344  LOG(1, "mm_free(%p)", ptr);
345  if (ptr == NULL) return;
346  assert(mm_initialized);
347 
348  ptr = PREV_PTR(ptr);
349 
350  LOG(1, "coalesce()");
351  TYPE bdrtag = PACK(GET_SIZE(ptr), FREE);
352  PUT(ptr, bdrtag);
353  PUT(END(ptr), bdrtag);
354 
355  // preceeding free block
356  if (ptr != heap_start && GET_STATUS(PREV_PTR(ptr)) == FREE) {
357  LOG(2, " coalescing with preceeding block");
358  void *prev_ptr = START(PREV_PTR(ptr));
359  size_t size = GET_SIZE(prev_ptr) + GET_SIZE(ptr);
360  TYPE bdrtag = PACK(size, FREE);
361  PUT(prev_ptr, bdrtag);
362  PUT(END(prev_ptr), bdrtag);
363  ptr = prev_ptr;
364  }
365 
366  // succeeding free block
367  if (NEXT_PTR(END(ptr)) != heap_end && GET_STATUS(NEXT_PTR(END(ptr))) == FREE) {
368  LOG(2, " coalescing with succeeding block");
369  void *next_ptr = NEXT_PTR(END(ptr));
370  TYPE bdrtag = PACK(GET_SIZE(next_ptr) + GET_SIZE(ptr), FREE);
371  PUT(ptr, bdrtag);
372  PUT(END(ptr), bdrtag);
373 
374  if (END(ptr) == PREV_PTR(heap_end)) {
375  LOG(1, "shrink_heap(0x%lx (%ld))", INITCHUNK, INITCHUNK);
376 
377  void *p = PREV_PTR(heap_end);
378  size_t size = (GET_STATUS(p) == FREE) ? GET_SIZE(p) : 0;
379  LOG(2, " last block");
380  LOG(2, " header %p", START(p));
381  LOG(2, " footer %p", p);
382  LOG(2, " size %lx (%ld)", GET_SIZE(p), GET_SIZE(p));
383  LOG(2, " status %d\n", GET_STATUS(p));
384 
385  size_t chunk_size = size > INITCHUNK ? ((size - INITCHUNK) / CHUNKSIZE) * CHUNKSIZE : 0;
386  if (chunk_size) {
387  LOG(2, " decrement heap by 0x%lx (%ld) bytes", chunk_size, chunk_size);
388  LOG(1, "ds_sbrk(-0x%lx)", chunk_size);
389 
390  void *last_block = START(PREV_PTR(heap_end));
391 
392  ds_sbrk(-chunk_size);
393  heap_end = ds_heap_brk = ds_sbrk(0);
394  heap_end = (TYPE*)(((TYPE)heap_end - 1) / BS * BS);
395  size = heap_end - last_block;
396 
397  TYPE bdrtag = PACK(size, FREE);
398  PUT(last_block, bdrtag);
399  PUT(PREV_PTR(heap_end), bdrtag);
400  PUT(heap_end, PACK(0, ALLOC));
401 
402  LOG(1, "ds_sbrk(+0x%lx)", 0);
403  LOG(2, " new heap_end at %p", heap_end);
404  LOG(2, " last block now at %p with size 0x%lx (%ld) bytes", last_block, size, size);
405  }
406  }
407  }

418 static void* ff_get_free_block(size_t size)
419 {
420  LOG(1, "ff_get_free_block(1x%lx (%lu))", size, size);
421  assert(mm_initialized);
422  void *p = heap_start;
423  int N = 1;
424 
425  LOG(2, " starting search at %p", p);
426  while (p < heap_end) {
427  LOG(2, " %p %d %d", p, GET_SIZE(p), GET_STATUS(p));
428  if ((GET_STATUS(p) == FREE) && (GET_SIZE(p) >= size)) break;
429  p += GET_SIZE(p);
430  N++;
431  }
432  if (p >= heap_end || GET_STATUS(p) == ALLOC) {
433  LOG(2, " %p %d %d", p, GET_SIZE(p), GET_STATUS(p));
434  LOG(1, " no suitable block found after %d tries.", N);
435  return NULL;
436  }
437  LOG(1, " --> match after %d tries.", N);
438  return p;
439 }

445 static void* nf_get_free_block(size_t size)
446 {
447  LOG(1, "nf_get_free_block(0x%x (%lu))", size, size);
448  assert(mm_initialized);
449  void *p = next_block;
450  int N = 0;
451  LOG(2, " starting search at %p", p);
452  do {
453  N++;
454  LOG(2, " %p %d %d", p, GET_SIZE(p), GET_STATUS(p));
455  if ((GET_STATUS(p) == FREE) && (GET_SIZE(p) >= size)) break;
456  p = GET_SIZE(p) ? p + GET_SIZE(p) : heap_start;
457  } while (p != next_block);
458  if (p == next_block && ((GET_STATUS(p) == ALLOC) || (GET_SIZE(p) < size))) {
459  LOG(1, " no suitable block found after %d tries.", N);
460  return NULL;
461  }
462  LOG(1, " --> match after %d tries.", N);
463  return next_block = p;
464 }

508  assert(mm_initialized);

510  void *p;
511  char *apstr;
512  if (get_free_block == ff_get_free_block) apstr = "first fit";
513  else if (get_free_block == nf_get_free_block) apstr = "next fit";
514  else if (get_free_block == bf_get_free_block) apstr = "best fit";
515  else apstr = "invalid";
516 
517  LOG(2, " allocation policy %s\n", apstr);
518  printf("\n----------------------------------------- mm_check ----------------------------------------------\n");

522  printf(" heap_end: %p\n", heap_end);
523  printf(" allocation policy: %s\n", apstr);
524  printf(" next_block: %p\n", next_block); // this will be needed for the next fit policy
525 
526  printf("\n");
527  p = PREV_PTR(heap_start);
528  printf(" initial sentinel: %p: size: %6lx (%7ld), status: %s\n",
529  p, GET_SIZE(p), GET_SIZE(p), GET_STATUS(p) == ALLOC ? "allocated" : "free");
530  p = heap_end;
531  printf(" end sentinel: %p: size: %6lx (%7ld), status: %s\n",
532  p, GET_SIZE(p), GET_SIZE(p), GET_STATUS(p) == ALLOC ? "allocated" : "free");
533  printf("\n");
534  printf(" blocks:\n");
535 
536  long errors = 0;
537  p = heap_start;
538  while (p < heap_end) {
539  TYPE hdr = GET(p);
540  TYPE size = SIZE(hdr);
541  TYPE status = STATUS(hdr);
542  printf(" %p: size: %6lx (%7ld), status: %s\n",
543  p, size, size, status == ALLOC ? "allocated" : "free");
544 
545  void *fp = p + size - TYPE_SIZE;
546  TYPE ftr = GET(fp);
547  TYPE fsize = SIZE(ftr);
548  TYPE fstatus = STATUS(ftr);
549 
550  if ((size != fsize) || (status != fstatus)) {
551  errors++;
552  printf(" --> ERROR: footer at %p with different properties: size: %lx, status: %lx\n",
553  fp, fsize, fstatus);
554  }
555 
556  p = p + size;
557  if (size == 0) {
558  printf(" WARNING: size 0 detected, aborting traversal.\n");
559  break;
560  }
561  }
562 
563  printf("\n");
564  if ((p == heap_end) && (errors == 0)) printf(" Block structure coherent.\n");
565  printf("-------------------------------------------------------------------------------------------------\n");
566 }
